8.1 游离事件插入算法
text
游戏启动时执行：
1. 确定场景列表 = [S-01, S-02]
2. 若难度 == 现实线，在 S-02 后插入 S-03
3. 确定可用空位 = 当前场景列表长度 - 1 个间隙
4. 对每个游离事件 (S-E1, S-E2, S-E3)：
   a. 以 30% 概率决定是否出场
   b. 若出场，在可用空位中随机选择一个位置插入
   c. 从可用空位中移除该位置
5. 添加 S-05/S-06（二选一，50%概率）
6. 添加 S-07 至末尾

function getESFJLine(sceneId, flags) {
  if (sceneId === "S-E1") {
    if (flags["F-01"] === "已触发") {
      return "……我在练习不替所有人填每一个空。有点不习惯。";
    } else {
      return "……我本来想说话的。但我在想，是不是可以不说。";
    }
  }
  if (sceneId === "S-E2") {
    if (flags["F-02"] === "已触发") {
      return "……那是第一次有人跟我说‘你组织得真好’。我留着它，不是因为那次聚会多好——是那句话让我觉得我做这些是值得的。";
    } else {
      return "……留着是因为我写了很久。后来很少写这样的东西了。";
    }
  }
  if (sceneId === "S-E3") {
    if (flags["F-02"] === "已触发" && flags["F-04"] === "已触发") {
      return "……今天有人让我帮忙做一件很花时间的事。我本来想答应。但我想了一下——我选不帮。";
    } else {
      return "……今天有人让我帮忙做一件很花时间的事。我不知道怎么拒绝。所以答应了。";
    }
  }
  return "";
}

function canUnlockPE1(flags) {
  return flags["Anchor-02"] && flags["Anchor-03"] && flags["Anchor-05"];
}

function canUnlockPE1_alt(flags) {
  return flags["Anchor-01"] && flags["Anchor-04"] && flags["Anchor-09"];

function canUnlockPE1_both(flags) {

function determineEnding(trust, flags, difficulty) {
  if (trust < 30) return "BE-1";
  if (trust >= 30 && trust < 50) return "BE-2";
  if (trust >= 50 && trust < 60) return "BE-3";
  if (trust >= 60 && trust < 70) return "NE-1";
  if (trust >= 70 && trust < 85) return "GE-1";
  if (trust >= 85 && difficulty === "现实线") {
    const count = Object.values(flags).filter(v => v === true).length;
    if (count >= 6 && flags["Anchor-02"] && flags["Anchor-05"] && flags["Anchor-06"]) {
      return "PE-1";
    }
    return "GE-1";
  }
  return "GE-1";
}
